  <script>
    function toggleAccordion(id, url) {
      const content = document.getElementById(id);
      const isVisible = content.style.display === "block";

      // Fecha todos os outros conteúdos
      document.querySelectorAll('.accordion-content').forEach(div => {
        div.style.display = 'none';
      });

      if (!isVisible) {
        // Se ainda não foi carregado, busca o conteúdo
        if (!content.dataset.loaded) {
          fetch(`./${url}`)
            .then(response => {
              if (!response.ok) throw new Error('Erro ao carregar o arquivo');
              return response.text();
            })
            .then(data => {
              content.innerHTML = data;
              content.dataset.loaded = "true";
              content.style.display = "block";
            })
            .catch(error => {
              content.innerHTML = `<p style="color:red;">Não foi possível carregar o conteúdo.</p>`;
              content.style.display = "block";
              console.error(error);
            });
        } else {
          content.style.display = "block";
        }
      }
    }
  </script>
