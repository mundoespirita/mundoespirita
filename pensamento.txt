    <script>
        function toggleAccordion(id, filename) {
            const content = document.getElementById(id);
            const isVisible = content.style.display === "block";

            // Esconde todos os conteúdos
            document.querySelectorAll('.accordion-content').forEach(c => c.style.display = 'none');

            if (!isVisible) {
                loadText(id, filename);
                content.style.display = "block";
                content.scrollIntoView({ behavior: 'smooth' });
            } else {
                content.style.display = "none";
            }
        }

        function loadText(id, filename) {
            fetch(filename)
                .then(response => {
                    if (!response.ok) {
                        throw new Error(`Erro ao carregar o arquivo: ${response.statusText}`);
                    }
                    return response.text();
                })
                .then(data => {
                    document.getElementById(id).innerHTML = data;
                })
                .catch(error => {
                    document.getElementById(id).innerHTML = `<p style="color:red;">Erro ao carregar o conteúdo.</p>`;
                    console.error('Erro ao carregar o texto:', error);
                });
        }
